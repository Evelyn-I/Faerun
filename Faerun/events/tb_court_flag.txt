namespace = courtflag

character_event = {
	id = courtflag.1
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		OR = {
			AND = {
				FROMFROM = {
					has_character_flag = vampire_court
				}
				OR = {
					trait = vampire
					trait = vampire_spawn
					trait = dhampyr
				}
			}
			AND = {
				FROMFROM = {
					has_character_flag = undead_court
				}
				OR = {
					trait = undead
					trait = lich
					trait = vampire
					trait = vampire_spawn
					trait = dhampyr
				}
			}
			AND = {
				FROMFROM = {
					has_character_flag = lycan_court
				}
				OR = {
					trait = wererat
					trait = werewolf
					trait = weretiger
				}
			}
		}
	}
	
	immediate = {
		if = {
			limit = {
				AND = {
					FROMFROM = {
						has_character_flag = vampire_court
					}
					OR = {
						trait = vampire
						trait = vampire_spawn
						trait = dhampyr
					}
				}
			}
			set_character_flag = vampire_court
		}
		if = {
			limit = {
				FROMFROM = {
					has_character_flag = undead_court
				}
				OR = {
					trait = undead
					trait = lich
					trait = vampire
					trait = vampire_spawn
					trait = dhampyr
				} 
			}
			set_character_flag = undead_court
		}
		if = {
			limit = {
				FROMFROM = {
					has_character_flag = lycan_court
				}
				OR = {
					trait = wererat
					trait = werewolf
					trait = weretiger
				}
			}
			set_character_flag = lycan_court
		}
	}
}
