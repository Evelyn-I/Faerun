is_native_outsider = {
	OR = {
		culture_group = planetouched_group
		culture = feyri
		culture = jann
	}
}

is_true_outsider = {
	OR = {
		culture_group = celestial_group
		culture_group = fiendish_group
		culture_group = gith_group
		AND = {
			culture_group = genie_group
			NOT = { culture = jann }
		}
	}
}
