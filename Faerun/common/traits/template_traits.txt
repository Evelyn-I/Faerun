natural_lycanthrope = {
	customizer = no
	is_visible = {
		OR = {
			character = FROM
			OR = {
				AND = {
					trait = werewolf
					FROM = { trait = werewolf }
				}
				AND = {
					trait = wererat
					FROM = { trait = wererat }
				}
				AND = {
					trait = weretiger
					FROM = { trait = weretiger }
				}
			}
		}
	}
	random = no
	opposites = { afflicted_lycanthrope }
}
